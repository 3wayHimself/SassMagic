   @include animation(
       (
           animationName: zoom,

               0: (

               100: (
                   transform: scale(3),
                   background-color: red

     )
   );

        0% {
            transform: scale(1);
            background-color: blue;
        }
        66.66666667% {
            transform: scale(3);
            background-color: red;
        }
        100% {
            transform: scale(3);
            background-color: red;
        }
    }
